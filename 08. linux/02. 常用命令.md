```bash
// ======================tree======================
// 生成树状结构
tree .
// 限制生成的层级，n表示层级数
tree -L n
// 正常显示中文，而不是显示转义字符
tree -N
// 将生成的树状图保存到一个txt文件中
tree . > tree.txt
```

```bash
// ======================nohup======================
// 不挂断运行命令
// 将test.out产生的所有日志输出到nohup.out中
nohup ./test.out &
// 只输出错误信息到日志文件
nohup ./test.out >/dev/null 2>log &
// 所有信息都不输出
nohup ./test.out >/dev/null 2>&1 & 
```
